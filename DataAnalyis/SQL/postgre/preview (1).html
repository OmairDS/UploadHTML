PostgreSQL for Data Analysis Practice

Dataset table name:

SQL 
dodo

Recommended PostgreSQL table structure:

SQL 
CREATE TABLE dodo (
    order_date DATE,
    city TEXT,
    rep TEXT,
    store TEXT,
    prod TEXT,
    cost NUMERIC,
    sale NUMERIC,
    profit NUMERIC
);
Note: If your Excel column is named Date, it is better to rename it to order_date in PostgreSQL.

10 Hard PostgreSQL Exercises

36. Rank cities by total sales

SQL 
SELECT 
    city,
    SUM(sale) AS total_sales,
    RANK() OVER (ORDER BY SUM(sale) DESC) AS sales_rank
FROM dodo
GROUP BY city;

37. Rank representatives inside each city

SQL 
SELECT 
    city,
    rep,
    SUM(sale) AS total_sales,
    RANK() OVER (
        PARTITION BY city 
        ORDER BY SUM(sale) DESC
    ) AS rep_rank_in_city
FROM dodo
GROUP BY city, rep
ORDER BY city, rep_rank_in_city;

38. Find the top product in each city

SQL 
WITH product_sales AS (
    SELECT 
        city,
        prod,
        SUM(sale) AS total_sales,
        RANK() OVER (
            PARTITION BY city 
            ORDER BY SUM(sale) DESC
        ) AS product_rank
    FROM dodo
    GROUP BY city, prod
)
SELECT *
FROM product_sales
WHERE product_rank = 1;

39. Calculate monthly sales with running total

SQL 
WITH monthly_sales AS (
    SELECT 
        DATE_TRUNC('month', order_date) AS month,
        SUM(sale) AS total_sales
    FROM dodo
    GROUP BY DATE_TRUNC('month', order_date)
)
SELECT 
    month,
    total_sales,
    SUM(total_sales) OVER (
        ORDER BY month
    ) AS running_total_sales
FROM monthly_sales
ORDER BY month;

40. Calculate month-over-month sales growth

SQL 
WITH monthly_sales AS (
    SELECT 
        DATE_TRUNC('month', order_date) AS month,
        SUM(sale) AS total_sales
    FROM dodo
    GROUP BY DATE_TRUNC('month', order_date)
),
monthly_growth AS (
    SELECT 
        month,
        total_sales,
        LAG(total_sales) OVER (ORDER BY month) AS previous_month_sales
    FROM monthly_sales
)
SELECT 
    month,
    total_sales,
    previous_month_sales,
    ((total_sales - previous_month_sales) / previous_month_sales) * 100 AS growth_percentage
FROM monthly_growth
WHERE previous_month_sales IS NOT NULL
ORDER BY month;

41. Find each store's contribution to total sales

SQL 
SELECT 
    store,
    SUM(sale) AS store_sales,
    SUM(sale) / SUM(SUM(sale)) OVER () * 100 AS percentage_of_total_sales
FROM dodo
GROUP BY store
ORDER BY percentage_of_total_sales DESC;

42. Find each product's contribution inside each city

SQL 
SELECT 
    city,
    prod,
    SUM(sale) AS product_sales,
    SUM(sale) / SUM(SUM(sale)) OVER (PARTITION BY city) * 100 AS percentage_of_city_sales
FROM dodo
GROUP BY city, prod
ORDER BY city, percentage_of_city_sales DESC;

43. Identify transactions above the average sale of their product

SQL 
SELECT 
    city,
    store,
    prod,
    sale,
    AVG(sale) OVER (PARTITION BY prod) AS avg_product_sale
FROM dodo
WHERE sale > (
    SELECT AVG(s2.sale)
    FROM dodo s2
    WHERE s2.prod = dodo.prod
);

44. Find the best representative per product

SQL 
WITH rep_product_sales AS (
    SELECT 
        prod,
        rep,
        SUM(sale) AS total_sales,
        RANK() OVER (
            PARTITION BY prod 
            ORDER BY SUM(sale) DESC
        ) AS rep_rank
    FROM dodo
    GROUP BY prod, rep
)
SELECT *
FROM rep_product_sales
WHERE rep_rank = 1;

45. Build a complete sales performance summary

SQL 
SELECT 
    city,
    store,
    prod,
    COUNT(*) AS total_orders,
    SUM(cost) AS total_cost,
    SUM(sale) AS total_sales,
    SUM(profit) AS total_profit,
    AVG(sale) AS avg_sale,
    AVG(profit) AS avg_profit,
    (SUM(profit) / SUM(sale)) * 100 AS profit_margin_percentage
FROM dodo
GROUP BY city, store, prod
ORDER BY total_profit DESC;
